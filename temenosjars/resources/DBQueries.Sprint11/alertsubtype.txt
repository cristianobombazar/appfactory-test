UPDATE dbxdb.alertsubtype 
SET 
    Status_id = 'SID_INACTIVE'
WHERE
    id != 'INTERNATIONAL_REQUEST_TO_INITIATOR'
        AND id != 'INTERNATIONAL_REQUEST_FOR_ALL_APPROVERS'
        AND id != 'RENOTIFY_PENDING_APPROVAL_REQUEST'
        AND id != 'INT_TRANSFER_RECIPIENT_ADDED'
        AND id != 'LOGIN_ATTEMPT'
        AND id != 'APPROVE_SINGLE_INTERNATIONAL_TRANSFER_INITIATOR'
        AND id != 'APPROVE_SINGLE_INTERNATIONAL_TRANSFER_APPROVERS'
        AND id != 'REJECT_SINGLE_INTERNATIONAL_TRANSFER_INITIATOR'
        AND id != 'REJECT_SINGLE_INTERNATIONAL_TRANSFER_APPROVERS'
        AND id != 'WITHDRAW_SINGLE_INTERNATIONAL_TRANSFER_INITIATOR'
        AND id != 'WITHDRAW_SINGLE_INTERNATIONAL_TRANSFER_APPROVERS'
        AND id != 'APPROVE_SINGLE_INTERNATIONAL_TRANSFER';
